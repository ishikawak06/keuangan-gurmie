const SPREADSHEET_ID = '1mRCc7UmU6gqV7n0RlO1s3lcEDhPoLAQX3mH-NsEjcCE'; function doGet(e) { try { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); let dbSheet = ss.getSheetByName('Database'); let catSheet = ss.getSheetByName('Categories'); // Buat sheet otomatis jika belum ada if (!catSheet) catSheet = ss.insertSheet('Categories'); if (!dbSheet) return createJsonResponse({ error: "Sheet 'Database' tidak ditemukan" }); // 1. Ambil Data Transaksi let transJson = []; const dbLastRow = dbSheet.getLastRow(); const dbLastCol = dbSheet.getLastColumn(); if (dbLastRow > 1 && dbLastCol > 0) { // Ambil semua data sekaligus (jauh lebih cepat) const dbData = dbSheet.getRange(1, 1, dbLastRow, dbLastCol).getValues(); const dbHeaders = dbData.shift(); // Pisahkan Header transJson = dbData.map(row => { let obj = {}; dbHeaders.forEach((h, i) => { if ((h === 'items' || h === 'reimburse') && row[i]) { try { obj[h] = JSON.parse(row[i]); } catch(e) { obj[h] = row[i]; } } else { obj[h] = row[i]; } }); return obj; }); } // 2. Ambil Data Kategori let catJson = []; const catLastRow = catSheet.getLastRow(); const catLastCol = catSheet.getLastColumn(); if (catLastRow > 1 && catLastCol > 0) { const catData = catSheet.getRange(1, 1, catLastRow, catLastCol).getValues(); const catHeaders = catData.shift(); catJson = catData.map(row => { let obj = {}; catHeaders.forEach((h, i) => obj[h] = row[i]); return obj; }); } // Kembalikan response berupa JSON yang rapi return createJsonResponse({ transactions: transJson, categories: catJson, ok: true }); } catch (err) { return createJsonResponse({ error: err.toString(), ok: false }); } } function doPost(e) { try { const payload = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.openById(SPREADSHEET_ID); let dbSheet = ss.getSheetByName('Database'); let catSheet = ss.getSheetByName('Categories'); // Proteksi pembuatan sheet otomatis if (!dbSheet) dbSheet = ss.insertSheet('Database'); if (!catSheet) catSheet = ss.insertSheet('Categories'); // ========================================== // 1. SIMPAN TRANSAKSI (BULK INSERT - SUPER CEPAT) // ========================================== if (payload.transactions && Array.isArray(payload.transactions)) { dbSheet.clear(); // Bersihkan data lama if (payload.transactions.length > 0) { let allKeys = new Set(); payload.transactions.forEach(item => Object.keys(item).forEach(key => allKeys.add(key))); const headers = Array.from(allKeys); // Siapkan array 2D untuk dimasukkan sekaligus const rowsToInsert = payload.transactions.map(item => { return headers.map(h => { let val = item[h]; if (val === undefined || val === null) return ""; if (typeof val === 'object') return JSON.stringify(val); return val; }); }); // Tulis Header & warnai agar rapi dbSheet.getRange(1, 1, 1, headers.length) .setValues([headers]) .setFontWeight("bold") .setBackground("#f3f4f6"); // Tulis Isi Data (1x Perintah untuk ratusan/ribuan baris) dbSheet.getRange(2, 1, rowsToInsert.length, headers.length).setValues(rowsToInsert); } } // ========================================== // 2. SIMPAN KATEGORI (BULK INSERT) // ========================================== if (payload.categories && Array.isArray(payload.categories)) { catSheet.clear(); // Bersihkan data lama if (payload.categories.length > 0) { let allKeys = new Set(); payload.categories.forEach(item => Object.keys(item).forEach(key => allKeys.add(key))); const headers = Array.from(allKeys); const rowsToInsert = payload.categories.map(item => { return headers.map(h => item[h] !== undefined ? item[h] : ""); }); // Tulis Header catSheet.getRange(1, 1, 1, headers.length) .setValues([headers]) .setFontWeight("bold") .setBackground("#f3f4f6"); // Tulis Data catSheet.getRange(2, 1, rowsToInsert.length, headers.length).setValues(rowsToInsert); } } // Kembalikan Response JSON agar mudah dibaca oleh Aplikasi Finance return createJsonResponse({ status: "Success", message: "Sinkronisasi Bulk sukses", ok: true }); } catch (err) { return createJsonResponse({ status: "Error", error: err.toString(), ok: false }); } } // Fungsi Helper Response JSON function createJsonResponse(data) { return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
